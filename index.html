Configure it


<property name="hibernate.search.default.indexmanager" value="elasticsearch"/>
<property name="hibernate.search.default.elasticsearch.host" value="http://127.0.0.1:9200"/>
<property name="hibernate.search.default.elasticsearch.index_schema_management_strategy" value="CREATE"/>
<property name="hibernate.search.default.elasticsearch.required_index_status" value="yellow"/>

Relations


@Entity
public class Movie {

	@Id
	Integer id;

	String title;

	@ManyToMany
	Set<Actor> actors = new HashSet<>();

}


@Indexed @Entity
public class Movie {

	@Id
	Integer id;

	@Field
	String title;

	@ManyToMany @IndexedEmbedded
	Set<Actor> actors = new HashSet<>();

}

Filters


List results = fullTextEntityManager
   .createFullTextQuery( query, Product.class )
   .enableFullTextFilter( "minorsFilter" )
   .list();


List results = fullTextEntityManager
   .createFullTextQuery( query, Product.class )
   .enableFullTextFilter( "minorsFilter" )
   .enableFullTextFilter( "specialDayOffers" )
      .setParameter( "day", “20151029” )
   .enableFullTextFilter( "inStockAt" )
      .setParameter( "location", "San Francisco" )
   .list();
